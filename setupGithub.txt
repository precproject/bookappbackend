A: Set Up GitHub Actions for React.js Deployment

[ON LOCAL MACHINE GENERATE PUBLIC KEY]
ssh-keygen -t ed25519
cat ~/.ssh/id_ed25519.pub

[COPY LOCAL GENERATED PUBLIC KEY ON VPS]
nano ~/.ssh/authorized_keys

chmod 700 ~/.ssh
chmod 600 ~/.ssh/authorized_keys

[VERIFY CONNECTING TO VPS FROM LOCAL IF SUCESS THEN COPY NEXT STEP PRIVATE KEY IN GITHUB]
ssh -i ~/.ssh/id_ed25519 user@your_vps_ip

[COPY LOCAL GENREATED PRIVATE KEY ON GIT]
cat ~/.ssh/id_ed25519

LOGIN TO GITHUB -> GitHub repository setting > Secrets and Variable > Actions. Choose "New Repository Secret"
CREATE CREDENTIALS -> VPS_HOST, VPS_USER(root), VPS_SSH_KEY

CREATE GITHUB WORKFLOW -> .github/workflows/react-deploy.yml
